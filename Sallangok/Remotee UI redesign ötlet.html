{
    internal static class Theme
    {
        public static Color AppBack = Color.FromArgb(245, 247, 250);
        public static Color Surface = Color.White;
        public static Color Sidebar = Color.FromArgb(240, 243, 248);
        public static Color Border = Color.FromArgb(220, 226, 234);

        public static Color Primary = Color.FromArgb(37, 99, 235);
        public static Color PrimaryLight = Color.FromArgb(219, 234, 254);

        public static Color Text = Color.FromArgb(17, 24, 39);
        public static Color Muted = Color.FromArgb(107, 114, 128);

        public static Color Success = Color.FromArgb(34, 197, 94);
        public static Color Danger = Color.FromArgb(239, 68, 68);

        public static Font FontRegular(float size = 9f)
            => new Font("Segoe UI", size, FontStyle.Regular);

        public static Font FontSemiBold(float size = 9f)
            => new Font("Segoe UI Semibold", size, FontStyle.Regular);

        public static Font FontBold(float size = 9f)
            => new Font("Segoe UI", size, FontStyle.Bold);

        public static void StylePrimaryButton(Button btn)
        {
            btn.FlatStyle = FlatStyle.Flat;
            btn.FlatAppearance.BorderSize = 0;
            btn.BackColor = Primary;
            btn.ForeColor = Color.White;
            btn.Font = FontSemiBold(9f);
            btn.Cursor = Cursors.Hand;
            btn.UseVisualStyleBackColor = false;
        }

        public static void StyleSecondaryButton(Button btn)
        {
            btn.FlatStyle = FlatStyle.Flat;
            btn.FlatAppearance.BorderSize = 1;
            btn.FlatAppearance.BorderColor = Border;
            btn.BackColor = Surface;
            btn.ForeColor = Text;
            btn.Font = FontSemiBold(9f);
            btn.Cursor = Cursors.Hand;
            btn.UseVisualStyleBackColor = false;
        }

        public static void StyleGhostButton(Button btn)
        {
            btn.FlatStyle = FlatStyle.Flat;
            btn.FlatAppearance.BorderSize = 0;
            btn.BackColor = Color.Transparent;
            btn.ForeColor = Muted;
            btn.Font = FontSemiBold(9f);
            btn.Cursor = Cursors.Hand;
            btn.UseVisualStyleBackColor = false;
        }

        public static void RoundControl(Control control, int radius = 14)
        {
            Rectangle rect = new Rectangle(0, 0, control.Width, control.Height);
            GraphicsPath path = new GraphicsPath();

            int d = radius * 2;
            path.StartFigure();
            path.AddArc(rect.X, rect.Y, d, d, 180, 90);
            path.AddArc(rect.Right - d, rect.Y, d, d, 270, 90);
            path.AddArc(rect.Right - d, rect.Bottom - d, d, d, 0, 90);
            path.AddArc(rect.X, rect.Bottom - d, d, d, 90, 90);
            path.CloseFigure();

            control.Region = new Region(path);
        }

        public static void AddHover(Button btn, Color normal, Color hover)
        {
            btn.MouseEnter += (s, e) => btn.BackColor = hover;
            btn.MouseLeave += (s, e) => btn.BackColor = normal;
        }

        public static void AddListBoxStyle(ListBox listBox)
        {
            listBox.BorderStyle = BorderStyle.None;
            listBox.BackColor = Surface;
            listBox.ForeColor = Text;
            listBox.Font = FontSemiBold(9.5f);
            listBox.IntegralHeight = false;
        }
    }
}

using System.Drawing;
using System.Windows.Forms;

namespace ManagementClient
{
    internal static class Theme
    {
        public static readonly Color AppBack = Color.FromArgb(245, 247, 250);
        public static readonly Color Surface = Color.White;
        public static readonly Color Sidebar = Color.FromArgb(240, 243, 248);
        public static readonly Color Border = Color.FromArgb(220, 226, 234);

        public static readonly Color Primary = Color.FromArgb(37, 99, 235);
        public static readonly Color PrimaryHover = Color.FromArgb(29, 78, 216);
        public static readonly Color PrimaryLight = Color.FromArgb(219, 234, 254);

        public static readonly Color Text = Color.FromArgb(17, 24, 39);
        public static readonly Color Muted = Color.FromArgb(107, 114, 128);
        public static readonly Color Success = Color.FromArgb(34, 197, 94);

        public static Font Regular(float size = 9f) => new Font("Segoe UI", size, FontStyle.Regular);
        public static Font Bold(float size = 9f) => new Font("Segoe UI", size, FontStyle.Bold);

        public static void StylePrimaryButton(Button btn)
        {
            btn.FlatStyle = FlatStyle.Flat;
            btn.FlatAppearance.BorderSize = 0;
            btn.BackColor = Primary;
            btn.ForeColor = Color.White;
            btn.Font = Bold(9f);
            btn.Cursor = Cursors.Hand;
            btn.UseVisualStyleBackColor = false;

            btn.MouseEnter += (s, e) => btn.BackColor = PrimaryHover;
            btn.MouseLeave += (s, e) => btn.BackColor = Primary;
        }

        public static void StyleSecondaryButton(Button btn)
        {
            btn.FlatStyle = FlatStyle.Flat;
            btn.FlatAppearance.BorderSize = 1;
            btn.FlatAppearance.BorderColor = Border;
            btn.BackColor = Surface;
            btn.ForeColor = Text;
            btn.Font = Bold(9f);
            btn.Cursor = Cursors.Hand;
            btn.UseVisualStyleBackColor = false;

            btn.MouseEnter += (s, e) => btn.BackColor = PrimaryLight;
            btn.MouseLeave += (s, e) => btn.BackColor = Surface;
        }

        public static void StyleGhostButton(Button btn)
        {
            btn.FlatStyle = FlatStyle.Flat;
            btn.FlatAppearance.BorderSize = 0;
            btn.BackColor = Color.Transparent;
            btn.ForeColor = Muted;
            btn.Font = Bold(9f);
            btn.Cursor = Cursors.Hand;
            btn.UseVisualStyleBackColor = false;
        }

        public static void StyleListBox(ListBox listBox)
        {
            listBox.BorderStyle = BorderStyle.None;
            listBox.BackColor = Surface;
            listBox.ForeColor = Text;
            listBox.Font = Bold(10f);
            listBox.IntegralHeight = false;
        }

        public static void EnableDoubleBuffer(Control control)
        {
            typeof(Control).GetProperty("DoubleBuffered",
                System.Reflection.BindingFlags.NonPublic | System.Reflection.BindingFlags.Instance)
                ?.SetValue(control, true, null);
        }
    }
}
